// Firestore Security Rules for HealthOps
// Copy and paste this into your Firestore Rules in Firebase Console

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Allow authenticated users to read/write their own tasks
    match /tasks/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Allow authenticated users to read/write their own notes
    match /notes/{document=**} {
      allow read, write: if request.auth != null;
    }

    // Allow authenticated users to read/write their own user info
    match /users/{userId} {
      allow read, write: if request.auth.uid == userId;
    }

    // Allow any authenticated user to read all users
    match /users/{document=**} {
      allow read: if request.auth != null;
    }
  }
}
